string removeDuplicates(string str) {
	    // code here
	    vector<char> A;
	    A.push_back(str[0]);
	    for(int i=1;i<str.length();i++)
	    {
	        if(find(A.begin(),A.end(),str[i])!=A.end())
	            continue;
	        else
	            A.push_back(str[i]);
	    }
	    string res="";
	    for(int i=0;i<A.size();i++)
	    {
	        res+=A[i];
	    }
	    return res;
	}
