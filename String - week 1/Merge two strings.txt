string merge (string S1, string S2)
{
    // your code here
    string res="";
    int i=0;
    while(i<S1.length() && i<S2.length())
    {
        res+=S1[i];
        res+=S2[i];
        i++;
    }
    while(i<S1.length())
        res+=S1[i++];
    while(i<S2.length())
        res+=S2[i++];
    return res;
}
