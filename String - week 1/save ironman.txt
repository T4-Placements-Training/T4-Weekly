bool saveIronman(string ch)
{
    // Complete the function
    string s="";
    for(int i=0;i<ch.length();i++)
    {
        if((ch[i]>64 && ch[i]<=90) || (ch[i]>=48 && ch[i]<=57))
            s+=ch[i];
        else if((ch[i]>96 && ch[i]<123))
            s+=ch[i]-32;
        
    }
    int n=s.length();
    string cmp="";
    for(int i=0;i<n/2;i++)
    {
        if(s[i]!=s[n-i-1])
            return false;
    }

    return true;
}
